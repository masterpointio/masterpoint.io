Place INSIDE a csi-section so the section provides the eyebrow + title; this emits just the question cards. Inner: blocks separated by `---`, each block: question: The question itself (short, ends with "?") body: The answer / commentary. Inline markdown works. */}} {{- $raw := .Inner -}} {{- $blocks := split $raw "---" -}}

" \t" -}} {{- if hasPrefix $line "question:" -}} {{- $question = trim (strings.TrimPrefix "question:" $line) " " -}} {{- else if hasPrefix $line "body:" -}} {{- $body = trim (strings.TrimPrefix "body:" $line) " " -}} {{- else if and $line $body -}} {{- $body = printf "%s %s" $body $line -}} {{- end -}} {{- end -}}
