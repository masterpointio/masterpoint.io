{{/* csi-questions — a takeaways row of numbered question cards ("ask these of your own platform"). Place INSIDE a csi-section so the section provides the eyebrow + title; this emits the compact question cards plus an optional emphasized outro line. Inner: blocks separated by `---`, each either: question: The question itself (short, ends with "?") body: The answer / commentary. Inline markdown works. or: outro: A closing line rendered large + centered under the cards. */}} {{- $raw := .Inner -}} {{- $blocks := split $raw "---" -}} {{- $outro := "" -}}
{{- $n := 0 -}} {{- range $blocks -}} {{- $b := trim . " \t\n" -}} {{- if $b -}} {{- $lines := split $b "\n" -}} {{- $question := "" -}} {{- $body := "" -}} {{- $mode := "" -}} {{- range $lines -}} {{- $line := trim . " \t" -}} {{- if hasPrefix $line "question:" -}} {{- $question = trim (strings.TrimPrefix "question:" $line) " " -}} {{- else if hasPrefix $line "body:" -}} {{- $body = trim (strings.TrimPrefix "body:" $line) " " -}} {{- $mode = "body" -}} {{- else if hasPrefix $line "outro:" -}} {{- $outro = trim (strings.TrimPrefix "outro:" $line) " " -}} {{- $mode = "outro" -}} {{- else if and $line (eq $mode "body") -}} {{- $body = printf "%s %s" $body $line -}} {{- else if and $line (eq $mode "outro") -}} {{- $outro = printf "%s %s" $outro $line -}} {{- end -}} {{- end -}} {{- if $question -}} {{- $n = add $n 1 -}}
{{ printf "%02d" $n }}

{{ $question | safeHTML }}

{{ $body | $.Page.RenderString (dict "display" "inline") }}

{{- end -}}
{{- end -}} {{- end -}} {{- end -}}

{{ . | $.Page.RenderString (dict "display" "inline") }}
