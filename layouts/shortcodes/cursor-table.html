{{- range $lineIndex, $line := $lines -}} {{- if eq $lineIndex 0 -}} {{- $row := trim $line " \t\r" -}} {{- $row = strings.TrimPrefix "|" $row -}} {{- $row = strings.TrimSuffix "|" $row -}} {{- range (split $row "|") }}{{ end -}} {{- end -}} {{- end }} {{- range $lineIndex, $line := $lines -}} {{- if gt $lineIndex 1 -}} {{- $row := trim $line " \t\r" -}} {{- $row = strings.TrimPrefix "|" $row -}} {{- $row = strings.TrimSuffix "|" $row -}} {{- range $cellIndex, $cell := (split $row "|") -}} {{- if eq $cellIndex 0 }} {{- else }}{{ end -}} {{- end }} {{- end -}} {{- end }}
{{ trim . " \t" | safeHTML }}
