{{/* csi-timeline — horizontal parallel-track bars for cutover / migration stories ("the old system keeps running while the new one comes up"). Place INSIDE a csi-section. Args: marker optional vertical marker position, percent 0–100 (e.g. "62") marker_label short label for the marker (e.g. "Cutover") Inner: blocks separated by `---`, one per track: label: track name (bold) note: short annotation shown after the label (muted) start: bar start percent (0–100) end: bar end percent (0–100) fade: "out" (old system winding down) | "in" (new system ramping up) | omit for solid */}} {{- $marker := .Get "marker" -}} {{- $markerLabel := .Get "marker_label" -}} {{- $raw := .Inner -}} {{- $blocks := split $raw "---" -}}
{{- if $marker -}}
{{- with $markerLabel }}{{ . | safeHTML }}{{ end -}}
{{- end -}} {{- range $i, $block := $blocks -}} {{- $b := trim $block " \t\n" -}} {{- if $b -}} {{- $lines := split $b "\n" -}} {{- $label := "" -}} {{- $note := "" -}} {{- $start := "0" -}} {{- $end := "100" -}} {{- $fade := "" -}} {{- range $lines -}} {{- $line := trim . " \t" -}} {{- if hasPrefix $line "label:" -}} {{- $label = trim (strings.TrimPrefix "label:" $line) " " -}} {{- else if hasPrefix $line "note:" -}} {{- $note = trim (strings.TrimPrefix "note:" $line) " " -}} {{- else if hasPrefix $line "start:" -}} {{- $start = trim (strings.TrimPrefix "start:" $line) " " -}} {{- else if hasPrefix $line "end:" -}} {{- $end = trim (strings.TrimPrefix "end:" $line) " " -}} {{- else if hasPrefix $line "fade:" -}} {{- $fade = trim (strings.TrimPrefix "fade:" $line) " " -}} {{- end -}} {{- end -}}
{{- if $label }}{{ $label | $.Page.RenderString (dict "display" "inline") }}{{ end -}} {{- if $note }}{{ $note | $.Page.RenderString (dict "display" "inline") }}{{ end -}}
{{- end -}} {{- end -}}
